long long fastPow(long long a,int b,long long m)
{
	long long ans = 1;
	while (b)
	{
		if (b & 1)
		{
			b--;
			ans *= a;
			ans %= m;
		}
		b >>= 1;
		a *= a;
		a %= m;
	}
	return ans;
}

	return fastPow(x, MOD - 2, MOD);
